320 std::string cleanedEvent = baseName.substr(0, baseName.length() - std::string("Event").length());
321 static_assert(std::is_base_of_v<IEvent, T>, "Class doesn't derive from IEvent");
322 luabridge::getGlobalNamespace(state)

325 if(!ref.isFunction())
326 return;
327
328 EventsManager::inst()->subscribe<T>(const_cast<luabridge::LuaRef&>(ref));
